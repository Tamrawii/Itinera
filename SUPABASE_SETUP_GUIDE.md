## Required Supabase Tables

### 1. Users Table
```sql
create table users (
  id bigint generated by default as identity primary key,
  email text unique not null,
  full_name text not null,
  role text not null check (role in ('tourist', 'provider', 'admin')),
  phone text,
  created_at timestamp with time zone default timezone('utc'::text, now()),
  updated_at timestamp with time zone default timezone('utc'::text, now())
);

-- Enable RLS
alter table users enable row level security;

-- Policies
create policy "Users can view their own profile" on users
  for select using (auth.uid()::text = id::text);

create policy "Admins can view all users" on users
  for select using (auth.jwt() ->> 'role' = 'admin');
```

### 2. Bookings Table
```sql
create table bookings (
  id bigint generated by default as identity primary key,
  tourist_id bigint references users(id),
  provider_id bigint references users(id),
  service_id bigint,
  booking_date timestamp with time zone not null,
  status text not null default 'pending' check (status in ('pending', 'confirmed', 'completed', 'cancelled')),
  total_price decimal(10,2) not null,
  created_at timestamp with time zone default timezone('utc'::text, now()),
  updated_at timestamp with time zone default timezone('utc'::text, now())
);

-- Enable RLS
alter table bookings enable row level security;

-- Policies
create policy "Tourists can view their own bookings" on bookings
  for select using (auth.uid()::text = tourist_id::text);

create policy "Providers can view bookings for their services" on bookings
  for select using (auth.uid()::text = provider_id::text);
```

### 3. Services Table
```sql
create table services (
  id bigint generated by default as identity primary key,
  provider_id bigint references users(id),
  name text not null,
  description text,
  price decimal(10,2) not null,
  category text not null,
  images text[],
  availability jsonb,
  location text,
  address text,
  created_at timestamp with time zone default timezone('utc'::text, now()),
  updated_at timestamp with time zone default timezone('utc'::text, now())
);

-- Enable RLS
alter table services enable row level security;
```

### 4. Reviews Table
```sql
create table reviews (
  id bigint generated by default as identity primary key,
  tourist_id bigint references users(id),
  service_id bigint references services(id),
  rating integer not null check (rating between 1 and 5),
  comment text,
  created_at timestamp with time zone default timezone('utc'::text, now()),
  updated_at timestamp with time zone default timezone('utc'::text, now())
);

-- Enable RLS
alter table reviews enable row level security;
```

### 5. Messages Table
```sql
create table messages (
  id bigint generated by default as identity primary key,
  sender_id bigint references users(id),
  receiver_id bigint references users(id),
  content text not null,
  read boolean default false,
  created_at timestamp with time zone default timezone('utc'::text, now())
);

-- Enable RLS
alter table messages enable row level security;
```

## Row Level Security (RLS) Policies

RLS is critical for security. Each table needs policies defining:
- Who can SELECT (view data)
- Who can INSERT (create data)
- Who can UPDATE (modify data)
- Who can DELETE (remove data)

Example pattern:
```sql
-- Allow users to only see their own data
create policy "Users can only access their own data" on table_name
  for all using (auth.uid()::text = user_id::text);

-- Allow admins to see all data
create policy "Admins have full access" on table_name
  for all using (auth.jwt() ->> 'role' = 'admin');
```
